// 通信与LNN的深度融合 impl LNN { /// 接收共振信号 pub fn receive_signal(&mut self, signal: &ResonanceSignal) -> SignalReaction { // 1. 检查信号是否与自己共振 let resonance_score = self.calculate_resonance(signal); // 2. 如果共振度高，产生强烈反应 if resonance_score > 0.8 { SignalReaction::Strong { attention: 1.0, learning_boost: 0.2, } } else if resonance_score > 0.5 { SignalReaction::Moderate { attention: 0.5, learning_boost: 0.1, } } else { SignalReaction::Weak { attention: 0.1, learning_boost: 0.0, } } } /// 发送共振信号 pub fn send_signal( &self, message: &str, channel: Channel, coins: &mut f64, ) -> Option<ResonanceSignal> { // 计算成本 let cost = channel.cost(message.len()); if *coins < cost { return None; // 金币不足，无法发送 } *coins -= cost; // 生成信号 Some(self.encoder.encode(message, channel.signal_type())) } }
