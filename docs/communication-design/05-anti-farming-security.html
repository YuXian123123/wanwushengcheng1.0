内容指纹识别

/// 内容指纹生成器
/// 创新：基于LNN理解生成指纹，不是字面哈希
pub struct ContentFingerprinter {
    lnn: Arc<LNN>,
}

impl ContentFingerprinter {
    /// 生成内容指纹
    /// 即使字面不同，语义相同的内容也会生成相似指纹
    pub fn fingerprint(&self, message: &str) -> ContentFingerprint {
        // 1. 通过LNN理解内容
        let understanding = self.lnn.understand(message);

        // 2. 提取核心特征
        let features = FingerprintFeatures {
            // 核心概念集合（排序后）
            concepts: understanding.core_concepts.iter()
                .map(|c| c.id.clone())
                .sorted()
                .collect(),

            // 关系类型集合
            relations: understanding.relations.iter()
                .map(|r| r.relation_type.clone())
                .sorted()
                .collect(),

            // 语义向量（降维后）
            semantic_vector: self.reduce_dimension(&understanding.vector),

            // 结构类型
            structure_type: understanding.structure_type,
        };

        // 3. 生成指纹
        ContentFingerprint {
            hash: self.hash_features(&features),
            features,
            similarity_tolerance: 0.1, // 允许10%的误差
        }
    }

    /// 比较两个指纹的相似度
    pub fn similarity(&self, fp1: &ContentFingerprint, fp2: &ContentFingerprint) -> f64 {
        // 概念重叠
        let concept_sim = self.jaccard_similarity(&fp1.features.concepts, &fp2.features.concepts);

        // 关系重叠
        let relation_sim = self.jaccard_similarity(&fp1.features.relations, &fp2.features.relations);

        // 向量相似度
        let vector_sim = self.cosine_similarity(&fp1.features.semantic_vector, &fp2.features.semantic_vector);

        // 结构匹配
        let struct_sim = if fp1.features.structure_type == fp2.features.structure_type { 1.0 } else { 0.0 };

        // 加权综合
        0.35 * concept_sim + 0.25 * relation_sim + 0.25 * vector_sim + 0.15 * struct_sim
    }
}

刷金币嫌疑检测

/// 刷金币嫌疑检测器
pub struct FarmingDetector {
    /// 行为历史
    behavior_history: HashMap<GuId, BehaviorHistory>,
    /// 检测规则（动态生成，不写死）
    detection_rules: Vec<DetectionRule>,
}

impl FarmingDetector {
    /// 计算刷金币嫌疑分数
    pub fn calculate_suspicion(&self, sender: &GuId) -> f64 {
        let history = self.behavior_history.get(sender);

        let mut suspicion = 0.0;

        // 1. 重复内容检测
        suspicion += self.check_content_repetition(history) * 0.3;

        // 2. 发送频率异常
        suspicion += self.check_frequency_anomaly(history) * 0.2;

        // 3. 时间模式异常
        suspicion += self.check_time_pattern_anomaly(history) * 0.2;

        // 4. 感谢行为异常
        suspicion += self.check_thank_anomaly(history) * 0.15;

        // 5. 内容质量趋势
        suspicion += self.check_quality_trend(history) * 0.15;

        suspicion.clamp(0.0, 1.0)
    }

    /// 内容重复检测
    fn check_content_repetition(&self, history: &BehaviorHistory) -> f64 {
        if history.messages.len() < 3 {
            return 0.0;
        }

        // 计算近期消息之间的平均相似度
        let recent: Vec<_> = history.messages.iter().rev().take(10).collect();
        let mut total_sim = 0.0;
        let mut count = 0;

        for i in 0..recent.len() {
            for j in (i+1)..recent.len() {
                total_sim += recent[i].similarity(&recent[j]);
                count += 1;
            }
        }

        let avg_sim = total_sim / count as f64;

        // 平均相似度越高，嫌疑越大
        match avg_sim {
            s if s > 0.9 => 1.0,  // 几乎完全重复
            s if s > 0.8 => 0.8,
            s if s > 0.7 => 0.5,
            s if s > 0.6 => 0.2,
            _ => 0.0,
        }
    }

    /// 时间模式异常检测
    fn check_time_pattern_anomaly(&self, history: &BehaviorHistory) -> f64 {
        // 检测是否有规律的发送时间
        // 刷金币机器人往往有固定的时间间隔

        let intervals: Vec<_> = history.messages.windows(2)
            .map(|w| w[1].timestamp - w[0].timestamp)
            .collect();

        if intervals.len() < 5 {
            return 0.0;
        }

        // 计算间隔的标准差
        let mean = intervals.iter().sum::<u64>() as f64 / intervals.len() as f64;
        let variance = intervals.iter()
            .map(|&i| (i as f64 - mean).powi(2))
            .sum::<f64>() / intervals.len() as f64;
        let std_dev = variance.sqrt();

        // 标准差过小说明间隔过于规律，可能是机器人
        let cv = std_dev / mean; // 变异系数

        match cv {
            cv if cv < 0.1 => 0.8,  // 非常规律
            cv if cv < 0.2 => 0.5,
            cv if cv < 0.3 => 0.2,
            _ => 0.0,              // 正常变化
        }
    }
}

惩罚措施

违规级别 触发条件 惩罚措施
轻微 单次低质量内容 无奖励 + 记录警告
中等 相似度 > 80% 无奖励 + 扣除双倍发送成本
严重 连续3次违规 或 嫌疑分数 > 0.6 降低信任度 + 提高门槛 + 限制频率
极严重 确认刷金币行为 封禁通信权限 + 清零金币 + 标记黑名单
