消息真实性验证

/// 消息验证器
pub struct MessageValidator {
    rules: Vec<ValidationRule>,
    history: MessageHistory,
}

impl MessageValidator {
    /// 四层验证机制
    pub fn validate(&self, signal: &ResonanceSignal) -> ValidationResult {
        // 第一层：身份验证
        let identity_check = self.verify_identity(signal)?;
        if !identity_check.passed {
            return ValidationResult::Reject("身份验证失败");
        }

        // 第二层：内容完整性
        let integrity_check = self.verify_integrity(signal)?;
        if !integrity_check.passed {
            return ValidationResult::Reject("内容完整性验证失败");
        }

        // 第三层：时序有效性
        let time_check = self.verify_timestamp(signal)?;
        if !time_check.passed {
            return ValidationResult::Reject("消息已过期");
        }

        // 第四层：行为合法性
        let behavior_check = self.verify_behavior(signal)?;
        if !behavior_check.passed {
            return ValidationResult::Reject("行为异常");
        }

        ValidationResult::Accept {
            confidence: identity_check.confidence
                      * integrity_check.confidence
                      * time_check.confidence
                      * behavior_check.confidence,
        }
    }
}

LNN共振指纹

/// LNN共振指纹生成
pub fn generate_resonance_fingerprint(lnn: &LNN) -> Fingerprint {
    let topology_hash = hash_topology(&lnn.connections);
    let weight_pattern = extract_weight_pattern(&lnn.weights);
    let response_signature = lnn.unique_response();

    Fingerprint {
        topology_hash,
        weight_pattern,
        response_signature,
        generated_at: now(),
        expires_at: now() + 3600, // 1小时有效期
    }
}

/// 验证发送者身份
pub fn verify_sender(
    signal: &ResonanceSignal,
    claimed_fingerprint: &Fingerprint,
) -> bool {
    // 1. 检查指纹是否过期
    if claimed_fingerprint.expires_at < now() {
        return false;
    }

    // 2. 验证共振模式是否匹配
    let expected_pattern = calculate_expected_pattern(&signal.source);
    if signal.pattern != expected_pattern {
        return false;
    }

    // 3. 验证响应签名
    signal.response_signature == claimed_fingerprint.response_signature
}

/// 通信权限定义
pub struct CommunicationPermission {
    pub world_broadcast: bool,
    pub create_group: bool,
    pub send_private: bool,
    pub rate_limit: RateLimit,
}

impl CommunicationPermission {
    /// 根据蛊虫状态动态调整权限
    pub fn from_creature(gu: &GuCreature) -> Self {
        Self {
            // 只有成熟期以上可以广播
            world_broadcast: gu.lifecycle.stage >= LifeStage::Adult,
            // 成长期以上可以创建小组
            create_group: gu.lifecycle.stage >= LifeStage::Larva,
            // 所有蛊虫可以私信
            send_private: true,
            // 根据金币余额调整频率
            rate_limit: RateLimit::from_coins(gu.coins),
        }
    }
}

通信审计与监控

⚠️ 审计要求：
  • 所有世界频道消息必须记录
  • 异常通信模式触发警报
  • 审计日志保留至少7天
  • 定期进行安全审计
/// 通信审计系统
pub struct CommunicationAudit {
    message_log: Vec<AuditRecord>,
    anomaly_detector: AnomalyDetector,
    alert_system: AlertSystem,
}

impl CommunicationAudit {
    /// 记录通信
    pub fn log(&mut self, signal: &ResonanceSignal, result: &ValidationResult) {
        self.message_log.push(AuditRecord {
            signal_id: signal.id,
            source: signal.source.clone(),
            channel: signal.channel,
            timestamp: now(),
            validation_result: result.clone(),
        });

        // 检测异常
        if let Some(anomaly) = self.anomaly_detector.detect(signal) {
            self.alert_system.raise_alert(anomaly);
        }
    }
}
