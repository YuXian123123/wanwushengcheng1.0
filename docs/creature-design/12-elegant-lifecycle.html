Ability 能力类型

/// 能力标识符 #[derive(Debug, Clone, PartialEq, Eq, Hash)] pub struct AbilityId(Uuid); /// 能力来源 #[derive(Debug, Clone)] pub enum AbilitySource { /// 从知识学习 Learned { knowledge_id: String, timestamp: u64 }, /// 从推理演化 Evolved { parent_ids: Vec<AbilityId> }, /// 从其他蛊虫继承 Inherited { from_gu: GuId, timestamp: u64 }, /// 组合创造 Combined { abilities: Vec<AbilityId> }, } /// 能力熟练度 p ∈ [0, 1] #[derive(Debug, Clone, Copy)] pub struct Proficiency(f64); impl Proficiency { pub fn new(value: f64) -> Self { Self(value.clamp(0.0, 1.0)) } /// 熟练度更新：p' = p + α(1-p)r - βp(1-r) pub fn update(&self, result: f64, learning_rate: f64) -> Self { let alpha = learning_rate; let beta = learning_rate * 0.5; let new_value = self.0 + alpha * (1.0 - self.0) * result - beta * self.0 * (1.0 - result); Self::new(new_value) } } /// 能力定义 #[derive(Debug, Clone)] pub struct Ability { pub id: AbilityId, pub name: String, pub source: AbilitySource, pub proficiency: Proficiency, pub concepts: HashSet<String>, pub created_at: u64, pub last_used: u64, pub usage_count: usize, } /// 能力容器（不可变操作） #[derive(Debug, Clone)] pub struct AbilityContainer { abilities: HashMap<AbilityId, Ability>, by_tag: HashMap<String, HashSet<AbilityId>>, combinations: Vec<CombinedAbility>, } impl AbilityContainer { /// 添加能力（返回新容器） pub fn with_ability(&self, ability: Ability) -> Self { let mut new_container = self.clone(); new_container.abilities.insert(ability.id.clone(), ability.clone()); // 更新标签索引 for concept in &ability.concepts { new_container.by_tag .entry(concept.clone()) .or_default() .insert(ability.id.clone()); } new_container } /// 更新熟练度（返回新容器） pub fn with_updated_proficiency( &self, ability_id: &AbilityId, result: f64, learning_rate: f64, ) -> Option<Self> { let ability = self.abilities.get(ability_id)?; let updated = Ability { proficiency: ability.proficiency.update(result, learning_rate), last_used: now(), usage_count: ability.usage_count + 1, ..ability.clone() }; Some(self.with_ability(updated)) } }

Lifecycle 生命周期类型

/// 生命周期状态 #[derive(Debug, Clone, Copy, PartialEq, Eq)] pub enum LifecyclePhase { /// 出生期 Birth, /// 成长期 Growth, /// 成熟期 Maturity, /// 衰老期 Aging, /// 死亡 Dead, } /// 生命周期配置 #[derive(Debug, Clone, Copy)] pub struct LifecycleConfig { /// 初始生命值 L₀ pub initial_lifespan: f64, /// 基础消耗率 r pub base_decay_rate: f64, /// 延寿成本 c₀ pub extension_cost: f64, /// 无延寿上限 - 物竞天择 } impl Default for LifecycleConfig { fn default() -> Self { Self { initial_lifespan: 1000.0, base_decay_rate: 1.0, extension_cost: 10.0, // 🔓 无 max_extensions，让自然选择决定 } } } /// 生命周期状态 #[derive(Debug, Clone)] pub struct Lifecycle { /// 剩余生命 L pub remaining: f64, /// 已延寿次数 n pub extensions: u32, /// 延寿历史 pub extension_history: Vec<ExtensionRecord>, /// 配置 config: LifecycleConfig, } impl Lifecycle { /// 创建新生命周期 pub fn new(config: LifecycleConfig) -> Self { Self { remaining: config.initial_lifespan, extensions: 0, extension_history: Vec::new(), config, } } /// 计算当前阶段 pub fn phase(&self) -> LifecyclePhase { let ratio = self.remaining / self.config.initial_lifespan; match ratio { r if r <= 0.0 => LifecyclePhase::Dead, r if r < 0.2 => LifecyclePhase::Aging, r if r < 0.5 => LifecyclePhase::Maturity, r if r < 1.0 => LifecyclePhase::Growth, _ => LifecyclePhase::Birth, } } /// 消耗生命（返回新生命周期） /// L' = L - r·Δt·f(LNN) pub fn with_decay(&self, dt: f64, lnn_health: f64) -> Self { let health_factor = match lnn_health { h if h >= 0.7 => 0.8, h if h >= 0.3 => 1.0, h if h >= 0.1 => 1.5, _ => 2.0, }; let decay = self.config.base_decay_rate * dt * health_factor; Self { remaining: (self.remaining - decay).max(0.0), ..self.clone() } } /// 延寿（返回新生命周期和成本） /// cost = c₀·amount/100 /// 🔓 无上限限制，物竞天择 pub fn with_extension(&self, amount: f64) -> Result<(Self, f64), ExtensionError> { let cost = self.config.extension_cost * (amount / 100.0); let mut new_lifecycle = self.clone(); new_lifecycle.remaining += amount; new_lifecycle.extensions += 1; new_lifecycle.extension_history.push(ExtensionRecord { amount, cost, timestamp: now(), }); // 聪明的蛊虫自然能活更久 Ok((new_lifecycle, cost)) } /// 是否存活 pub fn is_alive(&self) -> bool { self.remaining > 0.0 } }

MetaCognition 元认知类型

/// 自我评估 #[derive(Debug, Clone)] pub struct SelfAssessment { /// 自信程度 c ∈ [0,1] pub confidence: f64, /// 能力估计 ĉ ∈ [0,1] pub competence: f64, /// 知识缺口 pub knowledge_gaps: Vec<String>, } /// 监控指标 #[derive(Debug, Clone)] pub struct MonitoringMetrics { /// 推理准确率 pub reasoning_accuracy: f64, /// 学习效率 pub learning_efficiency: f64, /// 资源使用统计 pub resource_usage: ResourceStats, } /// 调节策略 #[derive(Debug, Clone)] pub enum RegulationStrategy { /// 调整推理深度 AdjustReasoningDepth { delta: i32 }, /// 切换学习焦点 SwitchLearningFocus { new_focus: String }, /// 优化资源分配 OptimizeResources { allocation: ResourceAllocation }, /// 进入休眠 EnterDormancy { duration: u64 }, } /// 反思记录 #[derive(Debug, Clone)] pub struct Reflection { /// 成功案例 pub successes: Vec<SuccessCase>, /// 失败案例 pub failures: Vec<FailureCase>, /// 洞察 pub insights: Vec<Insight>, } /// 元认知状态 #[derive(Debug, Clone)] pub struct MetaCognition { /// 自我评估 pub assessment: SelfAssessment, /// 监控指标 pub monitoring: MonitoringMetrics, /// 当前调节策略 pub current_strategy: Option<RegulationStrategy>, /// 反思记录 pub reflection: Reflection, /// 最后更新时间 pub last_update: u64, } impl MetaCognition { /// 执行自我监控（返回新元认知状态） pub fn with_monitoring(&self, metrics: MonitoringMetrics) -> Self { Self { monitoring: metrics, ..self.clone() } } /// 执行自我调节（返回新元认知状态） pub fn with_regulation(&self, strategy: RegulationStrategy) -> Self { Self { current_strategy: Some(strategy), ..self.clone() } } /// 记录反思（返回新元认知状态） pub fn with_reflection(&self, insight: Insight) -> Self { let mut new_reflection = self.reflection.clone(); new_reflection.insights.push(insight); Self { reflection: new_reflection, ..self.clone() } } /// 计算元认知得分 /// M = w₁·c + w₂·a + w₃·e pub fn score(&self) -> f64 { let w1 = 0.3; let w2 = 0.3; let w3 = 0.4; w1 * self.assessment.confidence + w2 * self.monitoring.reasoning_accuracy + w3 * self.monitoring.learning_efficiency } }

GuCreature 蛊虫实体

/// 蛊虫完整实体 #[derive(Debug, Clone)] pub struct GuCreature { /// 唯一标识 pub id: GuId, /// 名称 pub name: String, /// 生命周期 pub lifecycle: Lifecycle, /// 能力容器 pub abilities: AbilityContainer, /// 元认知 pub metacognition: MetaCognition, /// 认知币 pub coins: f64, /// 创建时间 pub created_at: u64, } impl GuCreature { /// 创建新蛊虫 pub fn new(id: GuId, name: String, config: LifecycleConfig) -> Self { Self { id, name, lifecycle: Lifecycle::new(config), abilities: AbilityContainer::new(), metacognition: MetaCognition::default(), coins: 100.0, created_at: now(), } } /// 学习能力（返回新蛊虫） pub fn with_learned_ability(&self, ability: Ability, cost: f64) -> Option<Self> { if self.coins < cost { return None; } Some(Self { abilities: self.abilities.with_ability(ability), coins: self.coins - cost, ..self.clone() }) } /// 消耗生命（返回新蛊虫） pub fn with_time_passed(&self, dt: f64, lnn_health: f64) -> Self { Self { lifecycle: self.lifecycle.with_decay(dt, lnn_health), ..self.clone() } } /// 延长生命（返回新蛊虫） pub fn with_extended_lifecycle(&self, amount: f64) -> Option<Self> { let (new_lifecycle, cost) = self.lifecycle.with_extension(amount).ok()?; if self.coins < cost { return None; } Some(Self { lifecycle: new_lifecycle, coins: self.coins - cost, ..self.clone() }) } /// 是否存活 pub fn is_alive(&self) -> bool { self.lifecycle.is_alive() } /// 获取元认知得分 pub fn metacognition_score(&self) -> f64 { self.metacognition.score() } }
