物竞天择机制

// 自然选择：聪明的蛊虫活得久
//
// 推理：
//   聪明的蛊虫 → 正确推理多 → 获得金币多 → 延寿次数多 → 活得久
//   愚蠢的蛊虫 → 错误推理多 → 金币消耗快 → 无法延寿 → 早死
//
// 结果：
//   长期存活的都是聪明的蛊虫
//   这就是"物竞天择，适者生存"

fn natural_selection(gu: &GuCreature) -> SurvivalOutcome {
    if gu.coins >= EXTENSION_COST {
        // 有金币，可以延寿
        SurvivalOutcome::Survive { extended: true }
    } else if gu.lifecycle.remaining > 0.0 {
        // 没金币但还有命
        SurvivalOutcome::Survive { extended: false }
    } else {
        // 金币耗尽，生命终结
        SurvivalOutcome::Die { reason: "金币耗尽" }
    }
}

// 计算实际生命消耗
fn actual_decay(lifecycle: &LifecycleState, lnn: &LNN) -> f64 {
    let base_rate = 1.0;

    // LNN健康度影响
    let health_factor = match lnn.health_score() {
        h if h >= 0.7 => 0.8,  // 健康，消耗降低
        h if h >= 0.3 => 1.0,  // 正常
        h if h >= 0.1 => 1.5,  // 不健康，消耗加速
        _ => 2.0,              // 危险，消耗翻倍
    };

    // 活跃度影响
    let activity_factor = if lnn.is_active() {
        1.2  // 活跃时消耗略高
    } else {
        0.5  // 休眠时消耗降低
    };

    base_rate * health_factor * activity_factor
}

金币延寿机制

// 延寿决策（无上限限制）
fn extend_lifecycle(
    lifecycle: &mut LifecycleState,
    coins: &mut CognitiveCoins,
    amount: u64,
) -> Result<u64, ExtensionError> {
    // 1. 计算金币成本
    let cost = calculate_extension_cost(amount);

    // 2. 检查金币是否足够
    if coins.balance() < cost {
        return Err(ExtensionError::InsufficientCoins);
    }

    // 3. 执行延寿
    coins.subtract(cost);
    lifecycle.remaining += amount as f64;
    lifecycle.extensions += 1;

    // 4. 记录延寿历史
    lifecycle.extension_history.push(ExtensionRecord {
        amount,
        cost,
        timestamp: now(),
    });

    // 💡 无延寿上限，物竞天择
    // 聪明的蛊虫自然能赚更多金币，活得更久

    Ok(amount)
}

// 延寿成本（固定成本，让能力决定生存）
fn calculate_extension_cost(amount: u64) -> f64 {
    let base_cost = 10.0; // 每100生命单位的成本
    base_cost * (amount as f64 / 100.0)
}

进化压力计算

// 计算蛊虫的"生存压力"
fn survival_pressure(gu: &GuCreature) -> f64 {
    // 生命剩余比例
    let life_ratio = gu.lifecycle.remaining / gu.lifecycle.initial_lifespan;

    // 金币与延寿成本比
    let coins_per_100_life = gu.coins / EXTENSION_COST;

    // 综合压力（越低越紧迫）
    let pressure = life_ratio * coins_per_100_life.min(1.0);

    pressure
}

// 压力驱动的行为选择
fn choose_action_by_pressure(gu: &GuCreature) -> Action {
    let pressure = survival_pressure(gu);

    match pressure {
        p if p < 0.1 => Action::Panic {  // 极度危险
            seek_coins_aggressively: true,
            risk_level: RiskLevel::High,
        },
        p if p < 0.3 => Action::Urgent {  // 紧急
            prioritize_coin_earning: true,
        },
        p if p < 0.5 => Action::Normal {  // 正常
            balance_learning_and_earning: true,
        },
        _ => Action::Comfortable {         // 舒适
            can_focus_on_learning: true,
        },
    }
}

死亡判定

enum DeathReason {
    NaturalAging,        // 自然衰老
    ResourceExhaustion,  // 资源耗尽
    LNNFailure,          // 神经网络崩溃
    SystemTermination,   // 系统终止
}

fn check_death(lifecycle: &LifecycleState, lnn: &LNN) -> Option<DeathReason> {
    // 检查生命值
    if lifecycle.remaining <= 0.0 {
        return Some(DeathReason::NaturalAging);
    }

    // 检查LNN崩溃
    if lnn.health_score() < 0.05 {
        return Some(DeathReason::LNNFailure);
    }

    // 检查资源耗尽
    if lifecycle.coins < 0.0 && lifecycle.remaining < 100.0 {
        return Some(DeathReason::ResourceExhaustion);
    }

    None
}
